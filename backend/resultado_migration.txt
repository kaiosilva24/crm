-- Migration: Create cache table for group participants
-- This allows fast sync and comparison with leads

CREATE TABLE IF NOT EXISTS group_participants_cache (
    id SERIAL PRIMARY KEY,
    phone VARCHAR(20) NOT NULL UNIQUE,
    name VARCHAR(255),
    source VARCHAR(50) DEFAULT 'baileys',
    groups TEXT[], -- Array of group names this participant belongs to
    last_synced_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_gpc_phone ON group_participants_cache(phone);
CREATE INDEX IF NOT EXISTS idx_gpc_synced ON group_participants_cache(last_synced_at);

COMMENT ON TABLE group_participants_cache IS 'Cached participants from all WhatsApp groups for fast sync';
COMMENT ON COLUMN group_participants_cache.groups IS 'Array of group names (e.g., {"Vendas 2024", "VIP"})';
